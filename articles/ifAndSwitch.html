    // Код выполнится, если условие истинно
}

Пример

int age = 18;

    System.out.println("Вы совершеннолетний.");
}

Условные конструкции в Java

    


    

        
1. Оператор if

        
if (условие) {
    // Код выполнится, если условие истинно
}


        
Пример

        
int age = 18;
if (age >= 18) {
    System.out.println("Вы совершеннолетний.");
}


    


    

        
2. Оператор if-else

        
if (условие) {
    // истинно
} else {
    // ложно
}


        
Пример

        
int age = 16;
if (age >= 18) {
    System.out.println("Вы совершеннолетний.");
} else {
    System.out.println("Вы несовершеннолетний.");
}


    


    

        
3. Оператор if-else if-else

        
if (условие1) {
    // ...
} else if (условие2) {
    // ...
} else {
    // ни одно условие не выполнилось
}


        
Пример

        
int age = 25;
if (age < 18) {
    System.out.println("Вы несовершеннолетний.");
} else if (age < 65) {
    System.out.println("Вы взрослый человек.");
} else {
    System.out.println("Вы пенсионер.");
}


    


    

        
4. Тернарный оператор

        
Краткая запись if-else

        
переменная = (условие) ? значение1 : значение2;


        
Пример

        
int age = 20;
String status = (age >= 18) ? "Совершеннолетний" : "Несовершеннолетний";
System.out.println(status);

    


    

        
5. Оператор switch

        
switch (переменная) {
    case значение1:
        // код
        break;
    case значение2:
        // код
        break;
    default:
        // если ничего не подошло
}


        
Пример

        
int day = 3;
switch (day) {
    case 1:
        System.out.println("Понедельник");
        break;
    case 2:
        System.out.println("Вторник");
        break;
    case 3:
        System.out.println("Среда");
        break;
    default:
        System.out.println("Некорректный день недели");
}


    


    

        
6. Вложенные условные операторы

        
int num = 10;
if (num > 0) {
    if (num % 2 == 0) {
        System.out.println("Число положительное и четное.");
    } else {
        System.out.println("Число положительное, но нечетное.");
    }
} else {
    System.out.println("Число не является положительным.");
}
